CREATE TABLE Customers(
	ID_customer INT,
	Surname VARCHAR(32),
	Location VARCHAR(64),
	Discount INT
);

CREATE TABLE Shops(
	ID_shop INT,
	Name VARCHAR(32),
	Location VARCHAR(64),
	Discount INT
);

CREATE TABLE Books(
	ID_book INT,
	Name VARCHAR(32),
	Cost INT,
	Storage VARCHAR(32),
	Count INT
);

CREATE TABLE Purchases(
	ID_purchase INT,
	Date_purchase DATE,
	ID_shop INT,
	ID_customer INT,
	ID_book INT,
	Count INT,
	Cost  INT
);
